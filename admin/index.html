-- ════════════════════════════════════════════════════════════════════ -- Adviuz — Multi-platform ad metrics (impressions etc.) Step 1 of 3 -- ONE ROW PER AD, holding its LATEST impression count (Make overwrites it -- each hour). Adviuz rolls rows up into platform totals and a grand -- "Total Digital Exposure" per campaign. -- Run in Supabase SQL editor. Safe to re-run. -- ════════════════════════════════════════════════════════════════════ create table if not exists ad_metrics ( id uuid primary key default gen_random_uuid(), campaign_id uuid references campaigns(id) on delete cascade, platform text not null, -- 'facebook','google','tiktok','linkedin','twitter',... ad_name text not null default '', -- platform ad name, e.g. 'FB Ad 1' external_ad_id text, -- platform ad id (optional, reference) external_campaign_id text, -- platform campaign id (optional, reference) impressions bigint default 0, -- latest total impressions for this ad clicks bigint default 0, spend numeric(12,2) default 0, reach bigint default 0, updated_at timestamptz default now(), source text default 'make' ); -- ONE row per (campaign, platform, ad). Make posts hourly with on_conflict -- on these 3 columns → the row is UPDATED to the latest number (no dupes). create unique index if not exists ad_metrics_uniq on ad_metrics (campaign_id, platform, ad_name); create index if not exists ad_metrics_campaign_idx on ad_metrics (campaign_id); -- Keep updated_at fresh on every overwrite. create or replace function adviuz_touch_ad_metrics() returns trigger as $$ begin new.updated_at = now(); return new; end; $$ language plpgsql; drop trigger if exists trg_touch_ad_metrics on ad_metrics; create trigger trg_touch_ad_metrics before update on ad_metrics for each row execute function adviuz_touch_ad_metrics(); -- ── Roll-up 1: per-platform totals (FB total, Google total, …) ── create or replace view campaign_platform_totals as select campaign_id, platform, sum(impressions) as impressions, sum(clicks) as clicks, sum(spend) as spend, sum(reach) as reach, count(*) as ad_count from ad_metrics group by campaign_id, platform; -- ── Roll-up 2: Total Digital Exposure per campaign (all platforms summed) ── create or replace view campaign_exposure as select campaign_id, sum(impressions) as total_impressions, sum(clicks) as total_clicks, sum(spend) as total_spend from ad_metrics group by campaign_id; -- ── RLS: a client can read metrics for their own campaigns ── alter table ad_metrics enable row level security; drop policy if exists ad_metrics_client_read on ad_metrics; create policy ad_metrics_client_read on ad_metrics for select to authenticated using ( exists ( select 1 from campaigns c join clients cl on cl.id = c.client_id where c.id = ad_metrics.campaign_id and cl.auth_user_id = auth.uid() ) ); -- NOTE: Make.com writes using the SERVICE ROLE key (bypasses RLS); the admin -- panel reads via the admin key. Mirror your existing campaigns RLS if different.
