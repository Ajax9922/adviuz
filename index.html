// ═══════════════════════════════════════════════════════ // Supabase Edge Function: process-reminders // Runs daily at 9am CST via pg_cron // Sends quote reminders and installment payment reminders // // PATCH (2026-06-06): removed the over-broad customer-level "stale quote" // guard. It was auto-expiring legitimate open quotes whenever the SAME // customer (by email) had ANY paid/accepted quote, or a newer quote. // That wrongly killed brand-new quotes for clients who'd paid an earlier, // unrelated quote. Each quote now stands on its own: it gets reminders on // its schedule and expires only when ITS OWN valid_until date passes. // Deliberate "replace the old quote" is still handled at send-time by the // cancel/keep prompt in the app (which sets the old quote to 'expired'). // // ALSO (2026-06-06): each quote reminder now writes a row to activity_log // (type 'reminder_sent') so it shows in the lead's activity timeline in // the hub, alongside "Quote Sent" / "Payment Received". // // ALSO (2026-06-07): a quote with reminders_off = true is skipped entirely, // so the hub's "Stop upcoming reminders" button can cancel them. // Requires: alter table public.quotes // add column if not exists reminders_off boolean not null default false; // ═══════════════════════════════════════════════════════ import { serve } from 'https://deno.land/std@0.168.0/http/server.ts' import { createClient } from 'https://esm.sh/@supabase/supabase-js@2' const SUPABASE_URL = Deno.env.get('SUPABASE_URL')! const SUPABASE_KEY = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')! const SEND_EMAIL_URL = `${SUPABASE_URL}/functions/v1/send-email` const APP_URL = Deno.env.get('APP_URL') || 'https://hub.adviuz.ca' const CORS = { 'Access-Control-Allow-Origin': '*' } serve(async (req) => { if (req.method === 'OPTIONS') return new Response('ok', { headers: CORS }) const supabase = createClient(SUPABASE_URL, SUPABASE_KEY) const today = new Date() today.setHours(0,0,0,0) let sent = 0, skipped = 0, errors = 0 // ── 1. QUOTE REMINDERS (7-day validity) ───────────── try { const { data: quotes } = await supabase .from('quotes') .select('*') .in('status', ['sent','viewed']) // only active quotes .not('valid_until', 'is', null) for (const q of (quotes || [])) { try { // Hub "Stop upcoming reminders" sets reminders_off = true → skip entirely. if (q.reminders_off) { skipped++; continue } const validUntil = new Date(q.valid_until) validUntil.setHours(0,0,0,0) const daysUntilExpiry = Math.round((validUntil.getTime() - today.getTime()) / 86400000) const daysSinceSent = Math.round((today.getTime() - new Date(q.sent_at || q.created_at).getTime()) / 86400000) // Reminder 1: Day 2 after sending (no response) if (daysSinceSent === 2 && !q.reminder_1_sent) { await sendQuoteReminder(supabase, q, 'friendly', daysUntilExpiry) await supabase.from('quotes').update({ reminder_1_sent: true }).eq('id', q.id) sent++ } // Reminder 2: 2 days before expiry else if (daysUntilExpiry === 2 && !q.reminder_2_sent) { await sendQuoteReminder(supabase, q, 'nudge', 2) await supabase.from('quotes').update({ reminder_2_sent: true }).eq('id', q.id) sent++ } // Reminder 3: Expiry day else if (daysUntilExpiry === 0 && !q.reminder_3_sent) { await sendQuoteReminder(supabase, q, 'urgent', 0) await supabase.from('quotes').update({ reminder_3_sent: true }).eq('id', q.id) sent++ } // Mark expired (day after expiry) — based on THIS quote's own date else if (daysUntilExpiry < 0 && q.status !== 'expired') { await supabase.from('quotes').update({ status: 'expired' }).eq('id', q.id) skipped++ } else { skipped++ } } catch (e) { console.error('[reminder] Quote error', q.id, (e as any).message) errors++ } } } catch (e) { console.error('[reminder] Quotes fetch failed:', (e as any).message) } // ── 2. INSTALLMENT REMINDERS ───────────────────────── try { const { data: installments } = await supabase .from('installment_payments') .select('*, quotes(recipient_name, recipient_email, recipient_biz, quote_number, recipient_province, status)') .eq('status', 'pending') for (const inst of (installments || [])) { try { const q = inst.quotes if (!q || !q.recipient_email) { skipped++; continue } // Don't chase installments on a quote that was retired/superseded. if (q.status === 'expired' || q.status === 'rejected' || q.status === 'cancelled') { skipped++; continue } const due = new Date(inst.due_date) due.setHours(0,0,0,0) const daysUntilDue = Math.round((due.getTime() - today.getTime()) / 86400000) const qUrl = `${APP_URL}/quote/?id=${inst.quote_id}` const fmtAmt = 'CAD $' + parseFloat(inst.amount).toLocaleString('en-CA',{minimumFractionDigits:2}) // 3 days before due if (daysUntilDue === 3 && !inst.reminder_1_sent) { await sendInstallmentReminder(supabase, inst, q, 'upcoming', fmtAmt, qUrl, daysUntilDue) await supabase.from('installment_payments').update({ reminder_1_sent: true }).eq('id', inst.id) sent++ } // Due today else if (daysUntilDue === 0 && !inst.reminder_2_sent) { await sendInstallmentReminder(supabase, inst, q, 'due_today', fmtAmt, qUrl, 0) await supabase.from('installment_payments').update({ reminder_2_sent: true }).eq('id', inst.id) sent++ } // 3 days overdue else if (daysUntilDue === -3 && !inst.reminder_3_sent) { await sendInstallmentReminder(supabase, inst, q, 'overdue', fmtAmt, qUrl, daysUntilDue) await supabase.from('installment_payments').update({ reminder_3_sent: true, status: 'overdue' }).eq('id', inst.id) sent++ } // 7 days overdue — final notice else if (daysUntilDue === -7 && !inst.reminder_4_sent) { await sendInstallmentReminder(supabase, inst, q, 'final_notice', fmtAmt, qUrl, daysUntilDue) await supabase.from('installment_payments').update({ reminder_4_sent: true }).eq('id', inst.id) sent++ } else { skipped++ } } catch (e) { console.error('[reminder] Installment error', inst.id, (e as any).message) errors++ } } } catch (e) { console.error('[reminder] Installments fetch failed:', (e as any).message) } console.log(`[process-reminders] sent=${sent} skipped=${skipped} errors=${errors}`) return new Response(JSON.stringify({ success: true, sent, skipped, errors }), { headers: { ...CORS, 'Content-Type': 'application/json' } }) }) // ── QUOTE REMINDER EMAILS ────────────────────────────── async function sendQuoteReminder( supabase: any, q: any, type: 'friendly' | 'nudge' | 'urgent', daysLeft: number ) { const qUrl = `${APP_URL}/quote/?id=${q.id}` const fmtAmt = 'CAD $' + parseFloat(q.total).toLocaleString('en-CA', {minimumFractionDigits:2}) const firstName = (q.recipient_name || 'there').split(' ')[0] const configs = { friendly: { subject: `Quick follow-up on your quote — ${q.quote_number}`, preview: `Just checking in — did you get a chance to review?`, headline: `Just checking in 👋`, body: `I wanted to make sure you received your quote and see if you have any questions.`, cta_color: '#0f1e2c', cta_text: 'Review Your Quote →', note: `Your quote is valid for ${daysLeft} more days.`, }, nudge: { subject: `Your quote expires in 2 days — ${q.quote_number}`, preview: `Don't miss out — your quote expires in 2 days`, headline: `Quote expires in 2 days ⏰`, body: `This is a quick reminder that your quote will expire in 2 days. To lock in the pricing, please review and accept before it expires.`, cta_color: '#d97706', cta_text: 'Accept Before It Expires →', note: `After expiry, a new quote may be required with updated pricing.`, }, urgent: { subject: `Last day — quote expires today — ${q.quote_number}`, preview: `Your quote expires today`, headline: `Quote expires today ⚠️`, body: `Today is the last day to accept your quote. After today, pricing may change and a new quote will be required.`, cta_color: '#dc2626', cta_text: 'Accept Now — Last Chance →', note: `If you need more time, simply reply to this email and we can extend it.`, }, } const cfg = configs[type] const html = buildReminderEmail({ firstName, headline: cfg.headline, body: cfg.body, quote_number: q.quote_number, amount: fmtAmt, valid_until: q.valid_until ? new Date(q.valid_until).toLocaleDateString('en-CA',{month:'long',day:'numeric',year:'numeric'}) : '', cta_url: qUrl, cta_text: cfg.cta_text, cta_color: cfg.cta_color, note: cfg.note, }) await fetch(SEND_EMAIL_URL, { method: 'POST', headers: { 'Authorization': `Bearer ${SUPABASE_KEY}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ to: q.recipient_email, subject: cfg.subject, html, type: 'transactional', from_email: 'sales@adviuz.com', from_name: 'Adviuz', reply_to: 'sales@adviuz.ca', lead_id: q.lead_id || null, }) }) // ── Log to the lead activity timeline so the operator can see reminders in-app. // Writes the same shape as the app's logActivity() → activity_log table. // Timeline maps type 'reminder_sent' to the ⏰ icon automatically. try { const noteByType: Record = { friendly: 'Follow-up reminder sent (checking in)', nudge: 'Reminder sent — quote expires in 2 days', urgent: 'Final reminder sent — quote expires today', } await supabase.from('activity_log').insert({ lead_id: q.lead_id || null, client_id: q.client_id || null, type: 'reminder_sent', label: '⏰ Quote Reminder', body: `${noteByType[type] || 'Reminder sent'} — ${q.quote_number || 'quote'}`, meta: { quote_id: q.id, quote_number: q.quote_number, reminder: type }, created_at: new Date().toISOString(), }) } catch (e) { console.error('[reminder] activity log failed', q.id, (e as any).message) } } // ── INSTALLMENT REMINDER EMAILS ──────────────────────── async function sendInstallmentReminder( supabase: any, inst: any, q: any, type: string, amount: string, qUrl: string, daysUntilDue: number ) { const firstName = (q.recipient_name || 'there').split(' ')[0] const dueDate = new Date(inst.due_date).toLocaleDateString('en-CA',{month:'long',day:'numeric',year:'numeric'}) const configs: Record = { upcoming: { subject: `Payment ${inst.installment_num} due in 3 days — ${q.quote_number}`, headline: `Payment coming up in 3 days 📅`, body: `Just a heads up — your installment payment is due in 3 days.
